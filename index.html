選擇排序法 (Selection Sort) - C 語言實作

這段程式碼展示了如何使用 選擇排序 演算法對整數陣列進行升序排序。其邏輯是每一輪從未排序的部分中找出最小值，並與當前位置交換。 
#include <stdio.h>

// 交換兩個整數的函式
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

// 選擇排序法主體
void selectionSort(int arr[], int n) {
    int i, j, min_idx;

    // 總共需要尋找 n-1 次最小值
    for (i = 0; i < n - 1; i++) {
        // 先假設目前的索引 i 就是最小值
        min_idx = i;

        // 從剩下的數字中找出真正的最小值索引
        for (j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }

        // 找到這輪最小的數字後，跟目前位置 i 交換
        if (min_idx != i) {
            swap(&arr[min_idx], &arr[i]);
        }
    }
}

// 印出陣列內容的函式
void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {
    int data[] = {29, 10, 14, 37, 13};
    int n = sizeof(data) / sizeof(data[0]);

    printf("原始陣列: ");
    printArray(data, n);

    selectionSort(data, n);

    printf("選擇排序結果: ");
    printArray(data, n);

    return 0;
}
